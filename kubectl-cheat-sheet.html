kubectl Cheat Sheet

Template Values Rendered & Copy
Setup
plugins=(git kubectl)
alias k=kubectl
 

      


      
      
1. Apply & Manage

      

        
k apply -f <file.yaml> -n <namespace>

        
          

          

        		
        

      

      

        
k scale deployment/<deployment-name> --replicas=<count> -n <namespace>

        
          

          

          

        		
        

      

      

        
k patch deployment/<deployment-name> -p '{"spec":{"replicas":<count>}}' -n <namespace>

        
          

          

          

        		
        

      


      
      
2. Inspect & List

      

        
k get pods -n <namespace>

        
          

        		
        

      

      

        
k get deployments -n <namespace>

        
          

        		
        

      

      

        
k get svc -n <namespace>

        
          

        		
        

      

      

        
k get nodes

        
        

      

      

        
k get ns

        
        

      

      

        
k get events -n <namespace>

        
          

        		
        

      


      
      
3. Describe & Explain

      

        
k describe pod/<pod-name> -n <namespace>

        
          

          

        		
        

      

      

        
k describe deployment/<deployment-name> -n <namespace>

        
          

          

        		
        

      

      

        
k describe node/<node-name>

        
          

        		
        

      

      

        
k explain deployment

        
        

      


      
      
4. Logs & Debug

      

        
k logs pod/<pod-name> -n <namespace>

        
          

          

        		
        

      

      

        
k logs -f pod/<pod-name> -n <namespace>

        
          

          

        		
        

      

      

        
k logs -f pod/<pod-name> -c <container-name> -n <namespace>

        
          

          

          

        		
        

      

      

        
k get deployment/<deployment-name> -o yaml -n <namespace>

        
          

          

        		
        

      

      

        
watch -n 2 'k get pods -n <namespace>'

        
          

        		
        

      


      
      
5. Exec & Port-Forward

      

        
k exec -it pod/<pod-name> -n <namespace> -- /bin/bash

        
          

          

        		
        

      

      

        
k port-forward svc/<service-name> <local-port>:<remote-port> -n <namespace>

        
          

          

          

          

        		
        

      


      
      
6. Rollouts & Updates

      

        
k rollout restart deployment/<deployment-name> -n <namespace>

        
          

          

        		
        
   
      
   
      

        
k rollout status deployment/<deployment-name> -n <namespace>

        
          

          

        		
        
   
      


      
      
7. Config & Context

      

        
kubectl config set-context --current --namespace=<namespace>

        
          

        		
        

      

      

        
kubectl config get-contexts

        
        

      

      

        
kubectl config use-context <context>

        
          

        		
        

      


      
      
8. Evict / Delete

      

        
kubectl evict pod/<pod-name> --namespace=<namespace>

        
          

          

        		
        

      

      

        
kubectl delete pod/<pod-name> --grace-period=0 --force --namespace=<namespace>

        
          

          

        		
        

      


      
      
9. Cluster Info & Metrics

      

        
k cluster-info

        
        

      

      

        
k top nodes

        
        

      

      

        
k top pods -n <namespace>

        
          

        		
        

      


      
      
10. File Copy

      

        
k cp <pod-name>:<source-path> <destination-path> -n <namespace>

        
          

          

          

          

        		
        

      

      

        
k cp <source-path> <pod-name>:<destination-path> -n <namespace>
